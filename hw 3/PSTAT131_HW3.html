Question 4

titanic_recipe <- recipe(survived ~ pclass, sex, age, sib_sp, parch, fare, data = titanic_train) %>%
  step_impute_linear() %>%
  step_dummy(all_nominal_predictors()) %>%
  step_interact(terms = ~ sex:fare +
                  age:fare)
